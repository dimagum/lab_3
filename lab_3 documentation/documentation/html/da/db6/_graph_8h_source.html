574std::pair<weight_t, route_t> dijkstra(const graph_t& graph, node_name_t key_from, node_name_t key_to) {
575 graph[key_from];
576 graph[key_to];
577
578
579 route_t route;
580 weight_t route_weight;
581
582 std::map<node_name_t, node_name_t> route_tmp;
583
584 const weight_t INF = std::numeric_limits<weight_t>::max();
585
586 route_tmp[key_from] = std::numeric_limits<node_name_t>::max();
587
588 std::map<node_name_t, weight_t> d;
589 std::map<node_name_t, bool> used;
590 for (auto [key, node] : graph) {
591 d[key] = INF;
592 used[key] = false;
593 }
594 d[key_from] = 0;
595
596 for (int i = 0; i < graph.size(); i++) {
597 node_name_t v = -1;
598
599
600
601 for (auto [key, node] : graph) {
602 if (!used[key] && (v == -1 || d[key] < d[v])) {
603 v = key;
604 }
605 }
606
607 if (d[v] == INF) {
608 break;
609 }
610 used[v] = true;
611
612 for (auto [to, len] : graph[v]) {
613 if ((std::numeric_limits<weight_t>::is_bounded) && len < 0) {
614 throw std::logic_error("there are negative weights in the graph.\n");
615 }
616 if (d[v] + len < d[to]) {
617 d[to] = d[v] + len;
618 route_tmp[to] = v;
619 }
620 }
621 }
622
623 if (route_tmp.find(key_to) == route_tmp.end()) {
624 throw std::logic_error("nodes are not connected.\n");
625 }
626
627 for (auto key = key_to; route_tmp[key] < std::numeric_limits<node_name_t>::max(); ) {
628 route.push_back(key);
629 key = route_tmp[key];
630 }
631
632 route.push_back(key_from);
633 std::reverse(route.begin(), route.end());
634
635 return std::pair<weight_t, route_t>(d[key_to], route);
636}
