559std::pair<weight_t, route_t> dijkstra(const graph_t& graph, node_name_t key_from, node_name_t key_to) {
560 graph[key_from];
561 graph[key_to];
562
563
564 route_t route;
565 weight_t route_weight;
566
567 std::map<node_name_t, node_name_t> route_tmp;
568
569 const weight_t INF = std::numeric_limits<weight_t>::max();
570
571 route_tmp[key_from] = std::numeric_limits<node_name_t>::max();
572
573 std::map<node_name_t, weight_t> d;
574 std::map<node_name_t, bool> used;
575 for (auto [key, node] : graph) {
576 d[key] = INF;
577 used[key] = false;
578 }
579 d[key_from] = 0;
580
581 for (int i = 0; i < graph.size(); i++) {
582 node_name_t v = -1;
583
584 for (auto [key, weight] : graph) {
585 if (!used[key] && (v == -1 || d[key] < d[v])) {
586 v = key;
587 }
588 }
589
590 if (d[v] == INF) {
591 break;
592 }
593 used[v] = true;
594
595 for (auto [to, len] : graph[v]) {
596 if (d[v] + len < d[to]) {
597 d[to] = d[v] + len;
598 route_tmp[to] = v;
599 }
600 }
601 }
602
603 for (auto key = key_to; route_tmp[key] < std::numeric_limits<node_name_t>::max(); ) {
604 route.push_back(key);
605 key = route_tmp[key];
606 }
607
608 route.push_back(key_from);
609 std::reverse(route.begin(), route.end());
610
611 return std::pair<weight_t, route_t>(d[key_to], route);
612}
